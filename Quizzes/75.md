def convert(input_str):
    binary_str = ""
    for characters in input_str:
        lett_bin = ord(characters)
        binary = ""
        while lett_bin > 0:
            res = lett_bin % 2
            binary = str(res) + binary
            lett_bin //= 2
        binary_str += binary
    return binary_str

#Test
test = "Hello World!"
binary_result = convert(test)
print(binary_result)
